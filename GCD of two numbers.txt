#include<bits/stdc++.h>
using namespace std;

class Solution {
  public:
    int gcd(int a, int b) {
        
        long long p=a,q=b;
       while (p!=q)
       {
           if(p>q)
           p=p-q;
           else
           q=q-p;
       }
       
        return p;
        
    }
};

int main(){
    int t;
    scanf("%d ",&t);
    while(t--){
        
        int a;
        scanf("%d",&a);
        
        
        int b;
        scanf("%d",&b);
        
        Solution obj;
        int res = obj.gcd(a, b);
        
        cout<<res<<endl;
        
    }
}
