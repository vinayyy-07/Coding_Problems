#include<bits/stdc++.h>
using namespace std;

class Solution
{
	public:
		long long int sum_of_square_evenNumbers(long long int n)
		{
		    // Code here
		    long long int m,s=0;
		    for(int i=1;i<=n;i++)
		    {
		        m=i*2;
		        s=s+m*m;
		    }
		    return s;
		}
};

int main(){
    int T;
    cin >> T;
    while(T--)
    {
    	long long int n;
    	cin >> n;
    	Solution ob;
    	long long int  ans = ob.sum_of_square_evenNumbers(n);
    	cout << ans <<"\n";
    }
	return 0;
}
