#include <bits/stdc++.h>
using namespace std;

class Solution {
  public:
    int nthTermOfAP(int a1, int a2, int n) {
        
        long n1=a1,n2=a2,p=n;
        long d=n2-n1,res;
        res=n1+(p-1)*d;
        return res;
        
        
    }
};

int main() {
    int t;
    cin >> t;
    while (t--) {
        int a1, a2, n;
        cin >> a1 >> a2 >> n;
        Solution ob;
        cout << ob.nthTermOfAP(a1, a2, n) << "\n";
    }
}
