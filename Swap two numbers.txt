
#include<bits/stdc++.h> 
using namespace std; 

class Solution{   
public:
    pair<int, int> get(int a, int b){
        pair<long, long> p;
        long long temp;
        temp=a;
        a=b;
        b=temp;
        p.first=a;
        p.second=b;
        return p;      
    }
};


int main() 
{ 
    int t;
    cin>>t;
    while(t--)
    {
        int a, b;
        cin >> a >> b;
        
        Solution ob;
        pair<int, int>p = ob.get(a, b);
        cout << p.first << ' ' << p.second << endl;
    
    }
    return 0; 
}
